YEAST (R)
Updated: 05 Oct 2018
Build code:
    Get raw data:
    http://uswest.ensembl.org/Saccharomyces_cerevisiae/Info/Index
        Download GTF
        Download genome FASTA
            Use unmasked chromosome files
            Terminal:
            for CHR in I II III IV V VI VII VIII IX X XI XII XIII XIV XV XVI Mito; do
                curl -O ftp://ftp.ensembl.org/pub/release-94/fasta/saccharomyces_cerevisiae/dna/Saccharomyces_cerevisiae.R64-1-1.dna.chromosome.${CHR}.fa.gz; done
        Curate rrna_list.fa
    STAR:
        STAR --runThreadN 2 --runMode genomeGenerate --genomeDir ../yeast_reference/genome \
            --genomeFastaFiles ~/scripts/ribopipe_deps/yeast_source/*.fa \
            --sjdbGTFfile ~/scripts/ribopipe_deps/Saccharomyces_cerevisiae.R64-1-1.94.gtf \
            --genomeSAindexNbases 11
            #--sjdbOverhang is x-1, where x is max read length of sequencing run
        STAR --runThreadN 2 --runMode genomeGenerate --genomeDir ../yeast_reference/rrna \
            --genomeFastaFiles ~/scripts/ribopipe_deps/yeast_rrna.fa \
    gtf2refFlat:
        https://bioconda.github.io/recipes/ucsc-gtftogenepred/README.html
        gtftogenepred Saccharomyces_cerevisiae.R64-1-1.94.gtf yeast_refFlat.txt
    metagene prep:
        metagene generate transcripts_cds_start --landmark cds_start --annotation_files transcripts.gtf --downstream 200


    Get yeast ncRNA fasta:
        ftp://ftp.ensembl.org/pub/release-94/fasta/saccharomyces_cerevisiae/ncrna/
    Make HISAT2 index:
        hisat2-build Saccharomyces_cerevisiae.R64-1-1.ncrna.fa ncrna_yeast_hisat2/ncrna
